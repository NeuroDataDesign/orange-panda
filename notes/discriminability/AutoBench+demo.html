Define the pre-processing method in a function which takes in trials and dictionaries of parameters and returns modified data trials.¶

E.g., in this sample, we define a function which takes in a EEG scan and returns the EEG scan with some PCA based denoising. We will also define a robust variant, which splits the data into a sum of a low rank and sparse matrix, and will continue with the sparse component thrown out.

In [4]:
from pcp import pcp
def pca_denoise(d, params):
    k = params['k'] # Number of singular vectors to take.
    s = params['s'] # Number of random data points to compute on.
    samp = np.random.choice(d.shape[1], s, replace=False)
    d_samp = d[:, samp]
    m = np.mean(d_samp, axis=1).reshape(-1, 1)
    d_cent = d_samp - m
    U, _, _ = np.linalg.svd(d_cent, full_matrices = False)
    P_k = U[:, :k].dot(U[:, :k].T)
    proj = P_k.dot(d)
    return proj

def robust_pca_denoise(d, params):
    k = params['k'] # Number of singular vectors to take.
    s = params['s'] # Number of random data points to compute on.
    samp = np.random.choice(d.shape[1], s, replace=False)
    d_samp = d[:, samp]
    d_samp, _ , _ = pcp(d_samp,
                        maxiter=50,
                        verbose=False, 
                        svd_method="approximate")
    m = np.mean(d_samp, axis=1).reshape(-1, 1)
    d_cent = d_samp - m
    U, _, _ = np.linalg.svd(d_cent, full_matrices = False)
    P_k = U[:, :k].dot(U[:, :k].T)
    proj = P_k.dot(d)
    return proj
